Đoạn mã Verilog minh họa FSM và bộ đếm:
\begin{lstlisting}[language=Verilog, caption={FSM Up/Down Counter – LAB2}]
module lab2_counter (
    input  wire       clk, rst_n,
    input  wire       up_down,   // 1=up, 0=down
    output reg  [7:0] count
);
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n)
            count <= 8'd0;
        else if (up_down)
            count <= count + 1'b1;
        else
            count <= count - 1'b1;
    end
endmodule

// ---- Clock Divider: 50 MHz -> 1 Hz ----
module clk_div_1hz (
    input  wire clk_in,   // 50 MHz
    input  wire rst_n,
    output reg  clk_1hz
);
    localparam HALF = 25_000_000;
    integer cnt;
    always @(posedge clk_in or negedge rst_n) begin
        if (!rst_n) begin cnt <= 0; clk_1hz <= 0; end
        else if (cnt == HALF - 1) begin cnt <= 0; clk_1hz <= ~clk_1hz; end
        else cnt <= cnt + 1;
    end
endmodule
\end{lstlisting}

Đoạn mã Verilog minh họa FSM I2C Master ở mức byte:
\begin{lstlisting}[language=Verilog, caption={I2C Master Control FSM (trích) – LAB3}]
// States
localparam IDLE      = 3'd0,
           START     = 3'd1,
           SEND_ADDR = 3'd2,
           ADDR_ACK  = 3'd3,
           TX_DATA   = 3'd4,
           DATA_ACK  = 3'd5,
           STOP      = 3'd6,
           DONE      = 3'd7;

reg [2:0] state, next_state;

// State register
always @(posedge clk or negedge rst_n) begin
    if (!rst_n) state <= IDLE;
    else        state <= next_state;
end

// Next-state logic
always @(*) begin
    case (state)
        IDLE:      next_state = start_req ? START : IDLE;
        START:     next_state = start_done ? SEND_ADDR : START;
        SEND_ADDR: next_state = byte_done ? ADDR_ACK : SEND_ADDR;
        ADDR_ACK:  next_state = ack_ok ? (rw ? RX_DATA : TX_DATA) : STOP;
        TX_DATA:   next_state = byte_done ? DATA_ACK : TX_DATA;
        DATA_ACK:  next_state = ack_ok ? (last_byte ? STOP : TX_DATA) : STOP;
        STOP:      next_state = stop_done ? DONE : STOP;
        DONE:      next_state = IDLE;
        default:   next_state = IDLE;
    endcase
end
\end{lstlisting}
