#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <inttypes.h>

#define SIZE 10

int main() {
    int16_t A[SIZE];
    int16_t B[SIZE];

    srand(time(NULL));

    // Tạo mảng A với giá trị ngẫu nhiên
    for (int i = 0; i < SIZE; i++) {
        A[i] = rand();
    }

    // Copy A sang B
    for (int i = 0; i < SIZE; i++) {
        B[i] = A[i];
    }

    // Sắp xếp B (bubble sort)
    for (int i = 0; i < SIZE - 1; i++) {
        for (int j = 0; j < SIZE - i - 1; j++) {
            if (B[j] > B[j + 1]) {
                int16_t temp = B[j];
                B[j] = B[j + 1];
                B[j + 1] = temp;
            }
        }
    }

    // In ra console mảng A
    printf("Array A (original):\n");
    for (int i = 0; i < SIZE; i++) {
        printf("A[%d] = %" PRId16 "\n", i, A[i]);
    }

    // In ra console mảng B
    printf("\nArray B (sorted):\n");
    for (int i = 0; i < SIZE; i++) {
        printf("B[%d] = %" PRId16 "\n", i, B[i]);
    }

    return 0;
}
