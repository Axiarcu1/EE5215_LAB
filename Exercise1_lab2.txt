#include	<stdio.h>
#include	<altera_avalon_pio_regs.h>
#include	<system.h>
#include	<unistd.h>

// Define BASE address
#define LED_BASE 0x81020
#define SW_BASE 0x81010
#define BTN_BASE 0x81000

#define	DELAY	500000
#define DEBOUNCE 100000
#define NONE_PRESSED 0xF


int	main() {
	printf("Hello	from	Nios	II!\n");
	int control = 0x01;
/*Loop	forever*/
	while	(1) {
		// SWITCH 1 ON
		if (IORD_ALTERA_AVALON_PIO_DATA(SW_BASE) == 0x01){
			for (int i = 0; i < 10; i++){
				IOWR_ALTERA_AVALON_PIO_DATA(LED_BASE,control);
				control = control << 1 | 0x01;
				usleep(DELAY);
			}
				control = 0x01;
				IOWR_ALTERA_AVALON_PIO_DATA(LED_BASE,0x00);
				usleep(DELAY);
		}
		// SWITCH 2 ON
		else if (IORD_ALTERA_AVALON_PIO_DATA(SW_BASE) == 0x02){
			control = 0x200;
			IOWR_ALTERA_AVALON_PIO_DATA(LED_BASE,0x00);
			usleep(DELAY);
			for (int i = 0; i < 10; i++){
				IOWR_ALTERA_AVALON_PIO_DATA(LED_BASE,control);
				control = control >> 1 | 0x200;
				usleep(DELAY);
			}
		}
		// Both SWITCH 1 and 2 ON/OFF
		else {
			IOWR_ALTERA_AVALON_PIO_DATA(LED_BASE,0x00);
		}
		usleep(DELAY);
	}
	return	0;
}
